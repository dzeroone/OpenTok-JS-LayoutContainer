1 /** @namespace Holds functionality for automatically managing the layout of OpenTok streams */
  2 var LayoutContainer = function() {
  3 	/** @private */
  4 	var Width;
  5 	
  6 	/** @private */
  7 	var Height;
  8 	
  9 	/** @private */
 10 	var containerId;	
 11 	
 12 	/** @scope LayoutContainer */
 13 	return {
 14 		/**
 15 		 * Initializes the LayoutContainer.  Must be called prior to any other functions.
 16 		 * @param {String} divId ID of DIV to be used as the container.
 17 		 * @param {int} width Width of container DIV.
 18 		 * @param {int} height Height of container DIV.
 19 		 */
 20 		init: function(divId, width, height){
 21 			containerId = divId;
 22 			Width = width;
 23 			Height = height;
 24 		},
 25 		
 26 		/**
 27 		 * Updates the container to incorporate any added or removed streams.
 28 		 */
 29 		layout: function(){
 30 			var subscriberBox = document.getElementById(containerId);
 31 			subscriberBox.style.position = "relative";
 32 			subscriberBox.style.width = Width + "px";
 33 			subscriberBox.style.height = Height + "px";
 34 			var vid_ratio = 3/4;
 35 
 36 			// Find ideal ratio
 37 			var count = subscriberBox.children.length;
 38 			var min_diff;
 39 			var targetCols;
 40 			var targetRows;
 41 			var availableRatio = Height / Width;
 42 			for (var i=1; i <= count; i++) {
 43 				var cols = i;
 44 				var rows = Math.ceil(count / cols);
 45 				var ratio = rows/cols * vid_ratio;
 46 				var ratio_diff = Math.abs( availableRatio - ratio);
 47 				if (!min_diff || (ratio_diff < min_diff)) {
 48 					min_diff = ratio_diff;
 49 					targetCols = cols;
 50 					targetRows = rows;
 51 				}
 52 			};
 53 
 54 			var videos_ratio = (targetRows/targetCols) * vid_ratio;
 55 
 56 			if (videos_ratio > availableRatio) {
 57 				targetHeight = Math.floor( Height/targetRows );
 58 				targetWidth = Math.floor( targetHeight/vid_ratio );
 59 			} else {
 60 				targetWidth = Math.floor( Width/targetCols );
 61 				targetHeight = Math.floor( targetWidth*vid_ratio );
 62 			}
 63 
 64 			var spacesInLastRow = (targetRows * targetCols) - count;
 65 			var lastRowMargin = (spacesInLastRow * targetWidth / 2);
 66 			var lastRowIndex = (targetRows - 1) * targetCols;
 67 
 68 			var firstRowMarginTop = ((Height - (targetRows * targetHeight)) / 2);
 69 			var firstColMarginLeft = ((Width - (targetCols * targetWidth)) / 2);
 70 			
 71 			var x = 0;
 72 			var y = 0;
 73 			for (i=0; i < subscriberBox.children.length; i++) {
 74 				if (i % targetCols == 0) {
 75 					// We are the first element of the row
 76 					x = firstColMarginLeft;
 77 					if (i == lastRowIndex) x += lastRowMargin;
 78 					y += i == 0 ? firstRowMarginTop : targetHeight;
 79 				} else {
 80 					x += targetWidth;
 81 				}
 82 				
 83 				var parent = subscriberBox.children[i];
 84 				var child = subscriberBox.children[i].firstChild;
 85 				
 86 				parent.style.position = "absolute";
 87 				parent.style.left = x + "px";
 88 				parent.style.top = y + "px";
 89 				
 90 				child.width = targetWidth;
 91 				child.height = targetHeight;
 92 				
 93 				parent.style.width = targetWidth + "px";
 94 				parent.style.height = targetHeight + "px";
 95 			};
 96 		},
 97 		/**
 98 		 * Adds a stream to the layout container
 99 		 * @param {String} divId The ID of the DIV container that is passed to session.subscribe()
100 		 */
101 		addStream: function(divId) {
102 			var container = document.createElement("div");
103 			var div = document.createElement("div");
104 			div.setAttribute('id', divId);
105 			container.appendChild(div);
106 			
107 			var subscriberBox = document.getElementById(containerId);
108 			subscriberBox.appendChild(container);			
109 		},
110 		
111 		/**
112 		 * Removes a stream from the layout container
113 		 * @param {String} subscriberId The ID of the subscriber object from the stream to be removed.
114 		 */
115 		removeStream: function(subscriberId) {
116 			var obj = document.getElementById(subscriberId);
117 			var container = obj.parentNode;
118 			container.parentNode.removeChild(container);
119 		}
120 	};
121 }();
